function createBook(title, author, pages) {
  return {
    title,
    author,
    pages,
    read() {
      console.log('Czytam książkę');
    }
  };
}

const book1 = createBook('Wiedźmin', 'Andrzej Sapkowski', 320);
const book2 = createBook('Clean Code', 'Robert Martin', 450);
